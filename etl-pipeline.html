Data Quality Framework

Built-in validation at each processing stage using Great Expectations:

# Data quality validation example
def validate_bronze_data(df: DataFrame) -> bool:
ge_df = ge.dataset.SparkDFDataset(df)
return all([
ge_df.expect_column_values_to_not_be_null("customer_id")["success"],
ge_df.expect_column_values_to_be_between("amount", 0, 10000)["success"]
])
